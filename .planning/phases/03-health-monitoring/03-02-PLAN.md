    **`ansible/playbooks/node-exporter.yml`:**
    ```yaml
    - name: Install and configure node_exporter on all monitored hosts
      hosts: monitored_hosts
      become: true
      pre_tasks:
        - name: Ensure Python3 is present (required for Ansible modules)
          raw: test -x /usr/bin/python3 || (apt-get update &amp;&amp; apt-get install -y python3)
          changed_when: false
      roles:
        - role: prometheus.prometheus.node_exporter
      post_tasks:
        - name: Verify node_exporter systemd service is active
          ansible.builtin.systemd:
            name: node_exporter
            state: started
            enabled: true
        - name: Probe :9100/metrics locally from each host
          ansible.builtin.uri:
            url: "http://127.0.0.1:9100/metrics"
            return_content: false
            status_code: 200
          register: ne_probe
        - name: Fail fast if metrics endpoint returns non-200
          ansible.builtin.fail:
            msg: "node_exporter on {{ inventory_hostname }} returned {{ ne_probe.status }}"
          when: ne_probe.status != 200
    ```
