    ```bash
    #!/usr/bin/env bash
    # tailscale-provision.sh — Install Tailscale and join the homelab mesh
    #
    # Usage (run on the target server as root):
    #   TAILSCALE_AUTH_KEY=tskey-auth-... bash tailscale-provision.sh [--hostname <name>] [--tags tag:homelab]
    #
    # Requirements:
    #   - Target must be Debian/Ubuntu/Fedora/RHEL-family (Tailscale installer handles detection)
    #   - Must be run as root (tailscale install needs apt/dnf + systemctl)
    #   - TAILSCALE_AUTH_KEY must be a valid ephemeral or reusable auth key from the Tailscale admin console
    #
    # This script is idempotent — running it twice on the same host is safe.

    set -euo pipefail

    # --- Argument parsing -----------------------------------------------------
    HOSTNAME_OVERRIDE=""
    TAGS=""
    ADVERTISE_ROUTES=""
    ACCEPT_ROUTES="--accept-routes"

    while [[ $# -gt 0 ]]; do
      case "$1" in
        --hostname) HOSTNAME_OVERRIDE="$2"; shift 2 ;;
        --tags)     TAGS="$2"; shift 2 ;;
        --advertise-routes) ADVERTISE_ROUTES="--advertise-routes=$2"; shift 2 ;;
        --no-accept-routes) ACCEPT_ROUTES=""; shift 1 ;;
        -h|--help)
          sed -n '1,20p' "$0"
          exit 0
          ;;
        *) echo "Unknown arg: $1" >&2; exit 2 ;;
      esac
    done

    # --- Pre-flight -----------------------------------------------------------
    if [[ -z "${TAILSCALE_AUTH_KEY:-}" ]]; then
      echo "ERROR: TAILSCALE_AUTH_KEY env var is required." >&2
      echo "Get one from: https://login.tailscale.com/admin/settings/keys" >&2
      exit 1
    fi

    if [[ "${TAILSCALE_AUTH_KEY}" != tskey-* ]]; then
      echo "ERROR: TAILSCALE_AUTH_KEY does not match tskey-* prefix; aborting." >&2
      exit 1
    fi

    if [[ $EUID -ne 0 ]]; then
      echo "ERROR: must be run as root." >&2
      exit 1
    fi

    # --- Install --------------------------------------------------------------
    if command -v tailscale >/dev/null 2>&1; then
      echo "tailscale already installed: $(tailscale version | head -1)"
    else
      echo "Installing Tailscale via official installer..."
      curl -fsSL https://tailscale.com/install.sh | sh
    fi

    # --- Join mesh ------------------------------------------------------------
    UP_ARGS=(--auth-key="${TAILSCALE_AUTH_KEY}")
    [[ -n "$ACCEPT_ROUTES" ]] && UP_ARGS+=("$ACCEPT_ROUTES")
    [[ -n "$HOSTNAME_OVERRIDE" ]] && UP_ARGS+=(--hostname="$HOSTNAME_OVERRIDE")
    [[ -n "$TAGS" ]] && UP_ARGS+=(--advertise-tags="$TAGS")
    [[ -n "$ADVERTISE_ROUTES" ]] && UP_ARGS+=("$ADVERTISE_ROUTES")

    echo "Joining Tailscale mesh..."
    tailscale up "${UP_ARGS[@]}"

    # --- Verify ---------------------------------------------------------------
    echo ""
    echo "Tailscale status:"
    tailscale status | head -20

    # Check this node has an IP
    TS_IP=$(tailscale ip -4 2>/dev/null | head -1)
    if [[ -z "${TS_IP}" ]]; then
      echo "ERROR: Tailscale reported no IPv4 address after `up`." >&2
      exit 1
    fi

    echo ""
    echo "OK — this node is ${TS_IP} on the mesh."
    ```

    Post-creation:
    ```bash
    chmod +x scripts/tailscale-provision.sh
    bash -n scripts/tailscale-provision.sh
    # If shellcheck is available (it's on cc-vk if installed):
    command -v shellcheck && shellcheck scripts/tailscale-provision.sh || true
    ```

    Critical safety rules embedded above:
    - `TAILSCALE_AUTH_KEY` MUST come from env var; script exits 1 if unset
    - Script validates the key has `tskey-` prefix before using it (format check, not auth check)
    - No auth key, no hostname, no tag is hardcoded
    - Exits non-zero if not root (install requires root)
    - Idempotent: if tailscale already installed, skips install
    - `set -euo pipefail` to fail fast on any error
