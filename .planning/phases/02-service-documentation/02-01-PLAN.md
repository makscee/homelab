    ```bash
    #!/usr/bin/env bash
    # verify-phase02.sh — Structural validation for Phase 02 (Service Documentation)
    # Covers SVC-01 through SVC-06. Runs in ~2 seconds. No SSH needed.
    set -uo pipefail

    ROOT="$(cd "$(dirname "$0")/.." && pwd)"
    FAIL=0
    pass() { echo "  ok: $1"; }
    fail() { echo "  FAIL: $1"; FAIL=1; }
    check_file() { if [[ -f "$ROOT/$1" ]]; then pass "$1 exists"; else fail "$1 missing"; fi; }
    check_grep() { if grep -qE "$2" "$ROOT/$1" 2>/dev/null; then pass "$1 contains /$2/"; else fail "$1 missing pattern /$2/"; fi; }
    check_no_grep() { if grep -qE "$2" "$ROOT/$1" 2>/dev/null; then fail "$1 contains forbidden pattern /$2/"; else pass "$1 free of /$2/"; fi; }

    echo "== SVC-01: docker-tower compose files with pinned tags =="
    for f in docker-compose.homestack.yml docker-compose.services.yml docker-compose.monitoring.yml; do
      check_file "servers/docker-tower/$f"
      # No :latest tag, no unpinned `image: name` without colon+version
      check_no_grep "servers/docker-tower/$f" ':latest[[:space:]]*$'
    done
    # Secrets must never be embedded in docker-tower compose files
    for f in servers/docker-tower/docker-compose*.yml; do
      [[ -f "$ROOT/$f" ]] || continue
      rel="${f#$ROOT/}"
      check_no_grep "$rel" '^[[:space:]]*(-|[A-Z_]+=)[^#]*(password|secret|api_key|token)[[:space:]]*[:=][[:space:]]*[A-Za-z0-9]'
    done
    check_file "servers/docker-tower/README.md"
    check_file "servers/docker-tower/.env.example"

    echo "== SVC-02: mcow systemd units + README =="
    check_file "servers/mcow/voidnet-bot.service"
    check_file "servers/mcow/voidnet-api.service"
    check_file "servers/mcow/README.md"
    check_grep "servers/mcow/README.md" '(STALE|DEPRECATED|stale|deprecated)'
    check_file "servers/mcow/.env.example"

    echo "== SVC-03: Proxmox LXC configs =="
    check_file "servers/tower/lxc-100-docker-tower.conf"
    check_file "servers/tower/lxc-101-tower-sat.conf"
    check_file "servers/tower/lxc-204-cc-vk.conf"
    check_file "servers/tower/README.md"

    echo "== SVC-04: AmneziaWG config encrypted =="
    # Either nether-awg.sops.yaml or nether.sops.yaml in hub-level secrets/
    if [[ -f "$ROOT/../secrets/nether-awg.sops.yaml" ]] || [[ -f "$ROOT/secrets/nether-awg.sops.yaml" ]] || [[ -f "$ROOT/../secrets/nether.sops.yaml" ]]; then
      pass "nether AWG sops file present (hub secrets/)"
    else
      fail "secrets/nether-awg.sops.yaml (or nether.sops.yaml) missing at hub level"
    fi
    check_file "servers/nether/README.md"
    check_file "servers/nether/docker-compose.services.yml"
    check_no_grep "servers/nether/docker-compose.services.yml" ':latest[[:space:]]*$'

    echo "== SVC-05: Tailscale provisioning script =="
    check_file "scripts/tailscale-provision.sh"
    if [[ -x "$ROOT/scripts/tailscale-provision.sh" ]]; then pass "tailscale-provision.sh executable"; else fail "tailscale-provision.sh not executable"; fi
    # bash syntax check
    if bash -n "$ROOT/scripts/tailscale-provision.sh" 2>/dev/null; then pass "tailscale-provision.sh parses"; else fail "tailscale-provision.sh syntax error"; fi
    # Auth key must come from env, not be hardcoded
    check_grep "scripts/tailscale-provision.sh" 'TAILSCALE_AUTH_KEY'
    check_no_grep "scripts/tailscale-provision.sh" 'tskey-[A-Za-z0-9]{10,}'

    echo "== SVC-06: tower-sat documented =="
    check_file "servers/tower-sat/README.md"
    # Either compose file or systemd equivalent documented in README
    if ls "$ROOT/servers/tower-sat"/docker-compose*.yml >/dev/null 2>&1 || grep -qiE 'systemd|no running services' "$ROOT/servers/tower-sat/README.md" 2>/dev/null; then
      pass "tower-sat has compose file or systemd/empty-state documented in README"
    else
      fail "tower-sat has no compose file and no systemd/empty-state note in README"
    fi

    echo "== Global: no plaintext private keys in repo =="
    if grep -rE 'BEGIN (RSA |OPENSSH |EC )?PRIVATE KEY|PrivateKey = [A-Za-z0-9+/=]{40,}' "$ROOT/servers" "$ROOT/scripts" 2>/dev/null | grep -v '.sops.yaml'; then
      fail "plaintext private key material found in tracked files"
    else
      pass "no plaintext private keys in servers/ or scripts/"
    fi

    if [[ $FAIL -eq 0 ]]; then
      echo ""
      echo "Phase 02 structural verification: PASS"
      exit 0
    else
      echo ""
      echo "Phase 02 structural verification: FAIL"
      exit 1
    fi
    ```
