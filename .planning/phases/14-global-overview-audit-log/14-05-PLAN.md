    (a) Create `apps/admin/app/api/alerts/count/route.ts`:

    ```typescript
    import "server-only";
    import { NextResponse } from "next/server";
    import { auth } from "@/auth";
    import { queryInstant } from "@/lib/prometheus.server";

    export const runtime = "nodejs";
    export const dynamic = "force-dynamic";

    const QUERY = `count by(severity)(ALERTS{alertstate="firing"})`;

    export async function GET() {
      const session = await auth();
      if (!session) return new NextResponse("unauthorized", { status: 401 });

      const bySeverity = { critical: 0, warning: 0, info: 0, other: 0 };
      let healthy = true;
      try {
        const samples = await queryInstant(QUERY);
        for (const s of samples) {
          const sev = (s.metric.severity ?? "").toLowerCase();
          const n = Number(s.value[1]);
          if (sev === "critical") bySeverity.critical += n;
          else if (sev === "warning") bySeverity.warning += n;
          else if (sev === "info") bySeverity.info += n;
          else bySeverity.other += n;
        }
      } catch {
        healthy = false;
      }
      const total = bySeverity.critical + bySeverity.warning + bySeverity.info + bySeverity.other;
      return NextResponse.json({ total, bySeverity, healthy });
    }
    ```

    `import "server-only"` is MANDATORY per DASH-05.

    (b) Create `apps/admin/app/(auth)/_components/useAlertCount.ts`:

    ```typescript
    import useSWR from "swr";

    export type AlertCount = {
      total: number;
      bySeverity: { critical: number; warning: number; info: number; other: number };
      healthy: boolean;
    };

    const fetcher = (url: string) => fetch(url).then(r => r.json());

    export function useAlertCount() {
      return useSWR<AlertCount>("/api/alerts/count", fetcher, {
        refreshInterval: 30_000,
        revalidateOnFocus: false,
        fallbackData: { total: 0, bySeverity: {critical:0,warning:0,info:0,other:0}, healthy: true },
      });
    }
    ```
