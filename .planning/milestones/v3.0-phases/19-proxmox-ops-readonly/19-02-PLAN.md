```ts
export const runtime = "nodejs";

export async function GET(req: NextRequest) {
  const session = await auth();
  if (!session?.user?.login) {
    return NextResponse.json({ error: "unauthorized" }, { status: 401 });
  }
  // (no CSRF for GETs per D-05/D-07; no logAudit per D-05 — read-only phase)
  // ...zod-validate path params if present...
  try {
    const data = await pveGet<...>('/path');
    return NextResponse.json({ data });
  } catch (e) {
    if (e instanceof PveError && e.code === "PVE_UNREACHABLE") {
      return NextResponse.json({ error: "tower unreachable", code: "PVE_UNREACHABLE" }, { status: 502 });
    }
    return NextResponse.json({ error: sanitizeErrorMessage(...) }, { status: 502 });
  }
}
```

PVE response envelope: `{ data: T }` — `pveGet` unwraps `.data`.
