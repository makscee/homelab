    export type StaleLevel = "fresh" | "stale" | "dead" | "unknown";

    export function classifyStale(
      lastSeenUnix: number | null,
      nowUnix: number = Date.now() / 1000,
    ): StaleLevel {
      if (lastSeenUnix == null) return "unknown";
      const age = nowUnix - lastSeenUnix;
      if (age < 90)  return "fresh";
      if (age < 300) return "stale";
      return "dead";

    export function buildHostRows(args: {
      cpu: PromSample[]; mem: PromSample[]; disk: PromSample[];
      containers: PromSample[]; uptime: PromSample[];
      load1: PromSample[]; load5: PromSample[]; load15: PromSample[];
      lastScrape: PromSample[];
      netRx: Record<string, number[]>;  // hostname → samples
      nowUnix?: number;
    }): HostRow[] {
      const nowUnix = args.nowUnix ?? Date.now() / 1000;
      const by = <K extends string>(samples: PromSample[]): Record<string, number> => {
        const out: Record<string, number> = {};
        for (const s of samples) {
          const inst = s.metric.instance;
          if (inst) out[inst] = Number(s.value[1]);
        }
        return out;
      };
      const cpuBy = by(args.cpu);
      const memBy = by(args.mem);
      const diskBy = by(args.disk);
      const uptimeBy = by(args.uptime);
      const load1By = by(args.load1);
      const load5By = by(args.load5);
      const load15By = by(args.load15);
      const lastBy = by(args.lastScrape);

      // Container count: cAdvisor `instance` uses IP:cadvisor_port; map by IP stem.
      const containerByName: Record<string, number> = {};
      for (const s of args.containers) {
        const ip = (s.metric.instance ?? "").split(":")[0];
        const name = CADVISOR_HOST_BY_IP[ip];
        if (name) containerByName[name] = Number(s.value[1]);
      }

      const rows: HostRow[] = [];
      for (const [instance, meta] of Object.entries(HOST_BY_INSTANCE)) {
        const lastSeen = lastBy[instance] ?? null;
        rows.push({
          name: meta.name,
          role: meta.role,
          cpuPct: instance in cpuBy ? cpuBy[instance] * 100 : null,
          memPct: instance in memBy ? memBy[instance] * 100 : null,
          diskPct: instance in diskBy ? diskBy[instance] * 100 : null,
          containerCount: meta.hasContainers ? (containerByName[meta.name] ?? null) : null,

          load5:  instance in load5By  ? load5By[instance]  : null,
          load15: instance in load15By ? load15By[instance] : null,
          netRx15m: args.netRx[meta.name] ?? [],
          lastScrapeUnix: lastSeen,
          stale: classifyStale(lastSeen, nowUnix),
        });
      }
      return rows;
    }

    // Helper: range samples as a flat number array for sparkline consumption.
    // Reuses queryRange; callers may omit if queryRange is sufficient.
    export async function queryRangeBytes(
      promql: string, start: number, end: number, stepSeconds: number,
    ): Promise<Record<string, number[]>> {
      const series = await queryRange(promql, start, end, stepSeconds);
      const out: Record<string, number[]> = {};
      for (const s of series) {
        const inst = s.metric.instance;
        if (!inst) continue;
        out[inst] = s.values.map(([, v]) => Number(v));
      }
      return out;
    }

    ```typescript
    import "server-only";
    import { NextRequest, NextResponse } from "next/server";
    import { auth } from "@/auth";
    import { queryInstant, queryRangeBytes } from "@/lib/prometheus.server";
    import { buildHostRows, HOST_BY_INSTANCE, CADVISOR_HOST_BY_IP } from "@/app/(auth)/_lib/overview-view-model";

    export const runtime = "nodejs";
    export const dynamic = "force-dynamic";

    const Q_CPU   = `1 - avg by(instance)(rate(node_cpu_seconds_total{mode="idle"}[2m]))`;
    const Q_MEM   = `1 - node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes`;
    const Q_DISK  = `1 - node_filesystem_avail_bytes{mountpoint="/",fstype!~"tmpfs|overlay"} / node_filesystem_size_bytes{mountpoint="/",fstype!~"tmpfs|overlay"}`;
    const Q_CONT  = `count by(instance)(container_last_seen{name!=""})`;
    const Q_UP    = `node_boot_time_seconds`;
    const Q_L1    = `node_load1`;
    const Q_L5    = `node_load5`;
    const Q_L15   = `node_load15`;
    const Q_LAST  = `timestamp(up{job="node"})`;
    const Q_NETRX = `sum by(instance)(rate(node_network_receive_bytes_total{device!~"lo|veth.*|docker.*|br-.*|tailscale.*"}[1m]))`;

    export async function GET(req: NextRequest) {
      const session = await auth();
      if (!session) return new NextResponse("unauthorized", { status: 401 });

      const now = Date.now() / 1000;
      const nowUnix = Math.floor(now);

      try {
        const [cpu, mem, disk, containers, uptimeRaw, load1, load5, load15, lastScrape, netRxMap] = await Promise.all([
          queryInstant(Q_CPU).catch(() => []),
          queryInstant(Q_MEM).catch(() => []),
          queryInstant(Q_DISK).catch(() => []),
          queryInstant(Q_CONT).catch(() => []),
          queryInstant(Q_UP).catch(() => []),
          queryInstant(Q_L1).catch(() => []),
          queryInstant(Q_L5).catch(() => []),
          queryInstant(Q_L15).catch(() => []),
          queryInstant(Q_LAST).catch(() => []),
          queryRangeBytes(Q_NETRX, nowUnix - 15 * 60, nowUnix, 30).catch(() => ({})),
        ]);

        // uptimeRaw gives boot time; translate to uptime seconds
        const uptime = uptimeRaw.map(s => ({
          metric: s.metric,
          value: [s.value[0], String(now - Number(s.value[1]))] as [number, string],
        }));

        // Map netRx instance→hostname for buildHostRows input
        const netRxByName: Record<string, number[]> = {};
        for (const [inst, samples] of Object.entries(netRxMap)) {
          const ip = inst.split(":")[0];
          // node_exporter instance = "IP:9100" → look up via HOST_BY_INSTANCE
          const meta = HOST_BY_INSTANCE[inst];
          if (meta) netRxByName[meta.name] = samples;
        }

        const hosts = buildHostRows({
          cpu, mem, disk, containers, uptime,
          load1, load5, load15, lastScrape,
          netRx: netRxByName,
          nowUnix: now,
        });

        return NextResponse.json({
          ts: Date.now(),
          prometheusHealthy: true,
          hosts,
        });
      } catch (err) {
        // This path only trips if something outside the per-query .catch() blew up
        // (e.g. auth throw, Node runtime issue). Signal full outage; let UI show banner.
        return NextResponse.json({
          ts: Date.now(),
          prometheusHealthy: false,
          hosts: [],
        }, { status: 200 });
      }
    }
    ```
