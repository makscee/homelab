      _sops_linux_amd64_sha256: "<LOOK UP AT AUTHORING TIME>"
        # Source: https://github.com/getsops/sops/releases/download/v3.9.4/sops-v3.9.4.checksums.txt
        # (sops-v3.9.4.linux.amd64 line; do NOT guess — read the release page)

    Stage 1b — sops + age:

      - name: Check current sops version on mcow
        ansible.builtin.command: /usr/local/bin/sops --version
        register: sops_installed
        failed_when: false
        changed_when: false

      - name: Download sops {{ _sops_version }} binary
        ansible.builtin.get_url:
          url: "https://github.com/getsops/sops/releases/download/v{{ _sops_version }}/sops-v{{ _sops_version }}.linux.amd64"
          dest: /usr/local/bin/sops
          mode: '0755'
          owner: root
          group: root
          checksum: "sha256:{{ _sops_linux_amd64_sha256 }}"
        when: sops_installed.rc != 0 or _sops_version not in (sops_installed.stdout | default(''))

      - name: Install age via apt (idempotent)
        ansible.builtin.apt:
          name: age
          state: present
          update_cache: true
          cache_valid_time: 3600

      - name: Verify sops binary on PATH as seen by a non-root shell
        ansible.builtin.command: /usr/local/bin/sops --version
        changed_when: false
        register: sops_verify
        failed_when: _sops_version not in sops_verify.stdout

    apt as `age` and that's the repo's stated preference (group_vars/all.yml
    implicit, see MEMORY 'mcow bun pinned' for the apt-first principle).
