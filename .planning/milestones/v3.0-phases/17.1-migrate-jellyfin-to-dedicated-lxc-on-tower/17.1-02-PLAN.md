    Step B — Create `ansible/playbooks/deploy-jellyfin.yml`. Minimum task set (copy shape from `deploy-docker-tower.yml`):
    ```yaml
    ---
    # Deploy Jellyfin native-deb on LXC 101 (Phase 17.1).
    # Idempotent; mirrors deploy-docker-tower.yml shape minus Docker tasks.
    - name: Deploy Jellyfin (native deb) on LXC 101
      hosts: jellyfin
      gather_facts: true
      become: true
      vars:
        jellyfin_packages:
          - jellyfin
          - jellyfin-ffmpeg7
          - intel-media-va-driver-non-free
          - libva-utils
      tasks:
        - name: Ensure keyrings dir
          ansible.builtin.file:
            path: /etc/apt/keyrings
            state: directory
            mode: '0755'

        - name: Jellyfin GPG key
          ansible.builtin.get_url:
            url: https://repo.jellyfin.org/jellyfin_team.gpg.key
            dest: /etc/apt/keyrings/jellyfin.gpg.key
            mode: '0644'

        - name: Jellyfin apt source
          ansible.builtin.copy:
            dest: /etc/apt/sources.list.d/jellyfin.sources
            mode: '0644'
            content: |
              Types: deb
              URIs: https://repo.jellyfin.org/debian
              Suites: bookworm
              Components: main
              Architectures: amd64
              Signed-By: /etc/apt/keyrings/jellyfin.gpg.key

        - name: Install Jellyfin + VA-API stack
          ansible.builtin.apt:
            name: "{{ jellyfin_packages }}"
            state: present
            update_cache: true

        - name: Hold Jellyfin packages (Ansible-managed upgrades only)
          ansible.builtin.dpkg_selections:
            name: "{{ item }}"
            selection: hold
          loop: "{{ jellyfin_packages }}"

        - name: Ensure jellyfin in render + video groups (D-08/D-10)
          ansible.builtin.user:
            name: jellyfin
            groups: render,video
            append: true
          notify: Restart jellyfin

        - name: Install transcode tmpfs mount unit
          ansible.builtin.template:
            src: templates/jellyfin-transcodes.mount.j2
            dest: /etc/systemd/system/var-cache-jellyfin-transcodes.mount
            mode: '0644'
          notify:
            - Reload systemd
            - Restart transcodes mount

        - name: Ensure transcode dir exists (pre-mount; mount hides it)
          ansible.builtin.file:
            path: /var/cache/jellyfin/transcodes
            state: directory
            owner: jellyfin
            group: jellyfin
            mode: '0755'

        - name: Enable + start tmpfs mount
          ansible.builtin.systemd:
            name: var-cache-jellyfin-transcodes.mount
            enabled: true
            state: started
            daemon_reload: true

        - name: Enable + start jellyfin
          ansible.builtin.systemd:
            name: jellyfin
            enabled: true
            state: started

        - name: Probe Jellyfin health
          ansible.builtin.uri:
            url: http://localhost:8096/health
            status_code: 200
          retries: 10
          delay: 3
          register: hc
          until: hc.status == 200

      handlers:
        - name: Reload systemd
          ansible.builtin.systemd: { daemon_reload: true }
        - name: Restart transcodes mount
          ansible.builtin.systemd:
            name: var-cache-jellyfin-transcodes.mount
            state: restarted
        - name: Restart jellyfin
          ansible.builtin.systemd:
            name: jellyfin
            state: restarted
    ```
    Per D-05: official repo. Per D-06: apt-mark hold (dpkg_selections). Per D-08: jellyfin in render+video groups (host GID visible via `dev0: gid=<HOST_RENDER_GID>` from Plan 01). Per D-09: systemd tmpfs unit.
