let cached: Set<string> | null = null;

export function getAllowedLogins(): Set<string> {
  if (cached) return cached;
  const raw = process.env.HOMELAB_ADMIN_ALLOWED_GITHUB_LOGINS ?? "";
  cached = new Set(

1. Create `apps/admin/middleware.ts`:
```ts
import NextAuth from "next-auth";
import { NextResponse } from "next/server";
import { authConfig } from "@/auth.config";
import { isLoginAllowed } from "@/lib/auth-allowlist";

const { auth } = NextAuth(authConfig);

const PUBLIC_PATHS = new Set<string>(["/login", "/403"]);
const isApiAuthPath = (p: string) => p.startsWith("/api/auth/");
const isStaticAsset = (p: string) =>
  p.startsWith("/_next/") || p === "/favicon.ico" || /\.[a-z0-9]+$/i.test(p);

function buildCsp(nonce: string): string {
  return [
    "default-src 'self'",
    `script-src 'self' 'nonce-${nonce}' 'strict-dynamic'`,
    "style-src 'self' 'unsafe-inline'",
    "img-src 'self' data: https://avatars.githubusercontent.com",
    "font-src 'self' data:",
    "connect-src 'self'",
    "frame-ancestors 'none'",
    "base-uri 'self'",
    "form-action 'self'",
  ].join("; ");
}

function applySecurityHeaders(res: NextResponse, nonce: string): NextResponse {
  res.headers.set("Content-Security-Policy", buildCsp(nonce));
  res.headers.set("Strict-Transport-Security", "max-age=63072000; includeSubDomains; preload");
  res.headers.set("X-Frame-Options", "DENY");
  res.headers.set("X-Content-Type-Options", "nosniff");
  res.headers.set("Referrer-Policy", "strict-origin-when-cross-origin");
  res.headers.set("Permissions-Policy", "camera=(), microphone=(), geolocation=()");
  return res;
}

export default auth((req) => {
  const { pathname } = req.nextUrl;
  const nonce = Buffer.from(crypto.getRandomValues(new Uint8Array(16))).toString("base64");

  const requestHeaders = new Headers(req.headers);
  requestHeaders.set("x-nonce", nonce);

  // Public paths + OAuth callback → always pass, but still apply headers.
  if (PUBLIC_PATHS.has(pathname) || isApiAuthPath(pathname) || isStaticAsset(pathname)) {
    const passRes = NextResponse.next({ request: { headers: requestHeaders } });
    return applySecurityHeaders(passRes, nonce);
  }

  const session = req.auth;
  if (!session) {
    const url = req.nextUrl.clone();
    url.pathname = "/login";
    return applySecurityHeaders(NextResponse.redirect(url), nonce);
  }

  const login = (session.user as { login?: string } | undefined)?.login;
  if (!isLoginAllowed(login)) {
    const url = req.nextUrl.clone();
    url.pathname = "/403";
    return applySecurityHeaders(NextResponse.redirect(url), nonce);
  }

  const res = NextResponse.next({ request: { headers: requestHeaders } });
  return applySecurityHeaders(res, nonce);
});

export const config = {
  // Match everything except Next internals + the API auth handler (which still needs headers applied via the public-path branch above).
  matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"],
};
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Caddy → Next.js | Untrusted HTTP request with optional session cookie enters at the Edge middleware |
| Auth.js → GitHub OAuth | Server-to-server OAuth callback with state+PKCE; GitHub replies with code+profile |
| Edge middleware → server component | Allowlist decision flows from env into request-processing code |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-04-01 | Spoofing | Session cookie forgery | mitigate | JWT signed with `AUTH_SECRET` (32+ bytes from SOPS); cookie is `__Secure-authjs.session-token` in prod with `httpOnly`+`secure`+`sameSite=lax` (SEC-07, Task 1 cookies config) |
| T-12-04-02 | Spoofing | Auth callback CSRF | mitigate | Auth.js v5 enables OAuth `state` param + PKCE by default (D-12-10, SEC-06); `trustHost: true` required because Caddy terminates TLS |
| T-12-04-03 | Elevation of Privilege | Non-allowlisted GitHub user reaches protected route | mitigate | Two-layer gate: (1) `signIn` callback in `auth.ts` rejects at OAuth completion — no session ever issued; (2) `middleware.ts` re-checks `isLoginAllowed(session.user.login)` on every request. Fail-closed default in `isLoginAllowed` (empty/missing login → false) |
| T-12-04-04 | Information Disclosure | RSC payload leaks server env into client bundle (P-03) | mitigate | `lib/auth-allowlist.ts` starts with `import "server-only"`; `eslint-plugin-server-only` installed (Task 1) and wired in Plan 05 lint gate; env vars never read in `"use client"` files |
| T-12-04-05 | Tampering | Clickjacking of /login or protected pages | mitigate | `X-Frame-Options: DENY` on every response; CSP `frame-ancestors 'none'` as secondary layer |
| T-12-04-06 | Tampering | Inline script injection | mitigate | CSP `script-src 'self' 'nonce-{per-request}' 'strict-dynamic'` — no `'unsafe-inline'` for scripts. Nonce generated in middleware, forwarded to server components via `x-nonce` request header |
| T-12-04-07 | Information Disclosure | Session cookie stolen via XSS | mitigate | `httpOnly: true` (SEC-07); CSP blocks inline script injection vector |
| T-12-04-08 | Repudiation | Session never expires — revocation impossible | accept | 8h `maxAge` with rolling refresh (SEC-07, D-12-08). Revocation via `AUTH_SECRET` rotation (acceptable for 1-operator allowlist per D-12-08). No DB sessions in Phase 12 (D-12-03). |
| T-12-04-09 | Spoofing | Tailscale header spoofing | n/a | OBSOLETED by GitHub OAuth switch (P-02 — documented as rationale for the switch) |

Residual risk: LOW. Remaining exposure is `AUTH_SECRET` compromise → full session forgery until rotation; mitigated by SOPS-encrypted storage (Plan 08) and 0600 file perms on mcow (Plan 09).
